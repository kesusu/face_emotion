### A3. 融合算法 (PC端设计好, 直接用)

```python
def ensemble_predict(self, features_dict, heatmap=None):
    """
    三模型加权融合
    
    参数:
        features_dict: extract_features()返回的特征字典
        heatmap: 可选, 如果没有就实时生成64×64热力图
        
    返回:
        {
            'primary': {'label': '开心', 'confidence': 0.62},
            'secondary': {'label': '惊讶', 'confidence': 0.25},
            'all_scores': {'惊讶':0.08,'恐惧':0.05,...'中性':0.12},
            'detail': {
                'svm': {'label':'开心','confidence':0.71},
                'cnn': {'label':'愤怒','confidence':0.38},
                'rule': {'label':'惊讶','score':6.5},
                'weights_used': {'svm':0.40,'cnn':0.35,'rule':0.25}
            }
        }
    """
    
    WEIGHTS = {'svm': 0.40, 'cnn': 0.35, 'rule': 0.25}
    LABELS = ['惊讶','恐惧','厌恶','开心','悲伤','愤怒','中性']
    
    # ── Step 1: 获取三路原始输出 ──
    
    # 1.1 SVM概率 (使用PC端新训v11模型)
    # 加载方式(在__init__中):
    #   import pickle
    #   with open('___AI_CNN训练交接包___/v11_svm_improved.pkl', 'rb') as f:
    #       svm_pkg = pickle.load(f)
    #   self.svm_model = svm_pkg['model']
    #   self.svm_scaler = svm_pkg['scaler']
    # 推理时:
    #   feat_vec = [features_dict.get(fn,0) for fn in svm_pkg['feature_names']]  # 14维
    #   feat_scaled = self.svm_scaler.transform([feat_vec])  # 必须标准化!
    #   svm_probs = self.svm_model.predict_proba(feat_scaled)[0]  # shape=(7,)
    
    # 1.2 CNN概率 (需实现CNN推理)
    if heatmap is None:
        heatmap = self._generate_heatmap(features_dict, size=64)
    cnn_probs = self._cnn_forward(heatmap)  # shape=(7,) softmax输出
    
    # 1.3 规则得分 (已有classify_emotion)
    rule_scores = classify_emotion_from_features(features_dict)  # dict{label: score}
    
    # ── Step 2: 归一化规则得分到概率分布 ──
    rule_vals = np.array([rule_scores.get(l, 0) for l in LABELS], dtype=float)
    rule_probs = softmax(rule_vals)  # 处理负值: exp(x-max)/sum
    
    # ── Step 3: 加权融合 ──
    fused = {}
    for i, label in enumerate(LABELS):
        fused[label] = (
            WEIGHTS['svm'] * svm_probs[i] +
            WEIGHTS['cnn'] * cnn_probs[i] +
            WEIGHTS['rule'] * rule_probs[i]
        )
    
    # ── Step 4: 排序取Top-K ──
    ranked = sorted(fused.items(), key=lambda x: x[1], reverse=True)
    
    return {
        'primary': {'label': ranked[0][0], 'confidence': round(ranked[0][1], 4)},
        'secondary': {'label': ranked[1][0], 'confidence': round(ranked[1][1], 4)},
        'all_scores': {k: round(v,4) for k,v in fused.items()},
        # ... detail信息
    }
```

### A4. 你需要实现的辅助方法

#### 方法1: `_generate_heatmap()` — 特征→64×64高斯热力图

参考 `cnn_heatmap_classifier_128.py` 中的 `generate_heatmap()` 函数:
```python
def _generate_heatmap(self, landmarks_468, size=64):
    """
    将468个MediaPipe关键点渲染为高斯热力图
    landmarks_468: shape=(468, 3), 归一化坐标
    返回: shape=(3, size, size), float32, [0,1]
    """
    import cv2
    heatmap = np.zeros((size, size), dtype=np.float32)
    
    sigma = size / 12.0  # 高斯半径
    coords = (landmarks_468[:, :2] * size).astype(np.int32)
    
    for (x, y) in coords:
        if 0 <= x < size and 0 <= y < size:
            # 绘制高斯点
            for dx in range(-int(sigma*2), int(sigma*2)+1):
                for dy in range(-int(sigma*2), int(sigma*2)+1):
                    nx, ny = x+dx, y+dy
                    if 0 <= nx < size and 0 <= ny < size:
                        g = np.exp(-(dx*dx+dy*dy)/(2*sigma*sigma))
                        heatmap[ny, nx] = max(heatmap[ny, nx], g)
    
    # 转3通道(RGB相同)
    heatmap_3ch = np.stack([heatmap]*3, axis=0)
    return heatmap_3ch.astype(np.float32)
```

#### 方法2: `_cnn_forward()` — CNN推理

```python
def _cnn_forward(self, heatmap):
    """CNN前向传播, 返回softmax概率"""
    import torch
    batch = torch.tensor(heatmap[np.newaxis,...], dtype=torch.float32).to(self.cnn_device)
    with torch.no_grad():
        logits = self.cnn_model(batch)
        probs = torch.softmax(logits, dim=1).cpu().numpy()[0]
    return probs
```

### 相比旧模型v8的改进

1. **class_weight='balanced'** — 解决多数类偏向问题(开心/中性不再占优)
2. **StandardScaler标准化** — 特征归一化, 提升数值稳定性
3. **C=1.0线性核调优** — CV选择最佳超参
4. **probability=True** — 支持predict_proba软输出(Ensemble必需)
5. **包含scaler打包** — 推理时自动标准化, 不需要额外处理
