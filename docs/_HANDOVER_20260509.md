# ① 热力图npy的正确路径和命名格式!
CACHE_DIR = 'RAF-DB/train/reports/heatmap_cache_128/'  # npy在根目录!
# 命名格式: {类别}_{jpg文件名}.npy
# 例: heatmap_cache_128/4悲伤_train_0442.jpg.npy

# ② npy内部是dict格式! (★ 新发现的坑 ★)

hm = data.item()['heatmap']  # dict{'heatmap': array(3,128,128), 'source': str}
# ❌ 直接np.load(npy_path) 会报错: Object arrays cannot be loaded

# ③ 完整加载代码:
npy_name = f"{cls_name}_{jpg}.npy"
npy_path = os.path.join(CACHE_DIR, npy_name)
data = np.load(npy_path, allow_pickle=True)
if data.shape == ():  # scalar object -> dict
    hm = data.item()['heatmap']
else:
    hm = data
